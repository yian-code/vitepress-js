ajax、跨域 #

XMLHttpRequest #

  • get请求方式：
// get请求
const xhr = new XMLHttpRequest(); 
xhr.open("GET", "/api", true) // true：异步请求
xhr.onreadystatechange = function () {
    if (xhr.readyState === 4) {
        if (xhr.status === 200) {
            console.log(JSON.parse(xhr.responseText));
        }
    }
}
xhr.send(null);
  • post请求方式：
const xhr = new XMLHttpRequest(); 
xhr.open("POST", "/api", true) // true：异步请求
xhr.onreadystatechange = function () {
    if (xhr.readyState === 4) {
        if (xhr.status === 200) {
            console.log(JSON.parse(xhr.responseText));
        }
    }
}
const postData = { name: 'yian', pwd: 'xxx'}; // 需要发送给服务器的数据
xhr.send(JSON.stringify(postData)); // 发送的是一个字符串

readyState和status #

XMLHttpRequest 的状态（state）会随着它的处理进度变化而变化。可以通过 xhr.readyState 来了解当前状态。

UNSENT = 0; // 初始状态
OPENED = 1; // open 被调用
HEADERS_RECEIVED = 2; // 接收到 response header
LOADING = 3; // 响应正在被加载（接收到一个数据包）
DONE = 4; // 请求完成

status是http请求状态码：

1xx   // http请求中
2xx   // 响应完成并成功返回    eg: 200
3xx   // 响应完成,资源重定向   eg: 301 302 304
4xx   // 响应发生客户端错误    eg: 404
5xx   // 响应发生服务端错误    eg: (有可能传参错误)

同源策略和跨域 #

源（origin）—— 域（domain）/端口（port）/协议（protocol）的组合。

跨源请求 —— 那些发送到其他域（即使是子域）、协议或端口的请求 —— 需要来自远程端的特殊 header。

这个策略被称为 “CORS”：跨源资源共享（Cross-Origin Resource Sharing）。

  • ajax请求时，浏览器要求当前网页和server必须同源（安全）
  • 同源：协议、域名、端口、三者一致即为同源

加载图片、CSS、JS可无视同源策略

 <img src="跨域图片地址" />  <!-- 可用于统计打点，可使用第三方统计服务  -->
<link href="跨域css地址" /> <!-- 可使用CDN，CDN一般都是跨域  -->
<script src="跨域js地址"></script> <!-- 可实现JSONP -->

跨域 #

  • 所有的跨域都必须经过server端允许和配合
  • 未经server端允许就实现跨域，说明浏览器有漏洞，这是危险的

在使用nodejs实现后端时，我们可以进行以下配置http header来实现跨域

server.on('request', function (req, res) {
    // 解决跨域问题   允许访问控制
    res.setHeader('Access-Control-Allow-Origin', '*')
    /** Access-Control-Allow-Origin 必须为 * 或
     * 进行请求的源（例如 https://javascript.info）才能允许此请求。
    **/
})

在使用Express时：

const express = require('express');
const app = express();

app.all('*', function(req, res, next) {    // 跨域
    res.setHeader('Access-Control-Allow-Origin', '*');
    next();
})

不建议直接写 * ，写跨域的域名称即可

JSONP #

  • <script>可绕过跨域
  • 服务器可以任意动态拼接数据返回
  • 所以，<script>就可以获得跨域的数据，只要服务端愿意返回
<script>
    window.abc = function (data) {
        console.log(data);
    }
</script>

<script src="http://localhost:8882/jsonp.js?name=xxx&callback=abc">
</script>

<!-- jsonp.js 
abc(
    { name: 'xxx' }
) 
-->

使用jQuery实现jsonp：

$.ajax({
    url: 'http://localhost:8882/jsonp.js',
    dataType: 'jsonp',
    jsonpCallback: 'callback',
    success: function(data) {
        console.log(data);
    }
})

任务：手写ajax #

function ajax(url) = {
    const p = new Promise( (resolve, reject) => {
        const xhr = new XMLHttpRequest();
        xhr.open('GET', url, true);
        xhr.onreadystatechange = function() {
            if(xhr.readyState === 4) {
                if(xhr.status === 200) {
                    resolve(
                        JSON.parse(xhr.responseText);
                    )
                } else if(xhr.status === 404){
                    reject( new Error('404 not found'));
                } else {
                    ....
                }
            }
        }
        xhr.send(null);
    })
    return p;
}

const dataUrl = '/data/test.json';

ajax(dataUrl)
.then(res => console.log(res))
.catch(err => console.log(err))

vue跨域 #

  1. 在vue.config.js中设置
module.exports = {
    dev: {
        // Paths
        assetsSubDirectory: 'static',
        assetsPublicPath: '/',
        proxyTable: { // 配置跨域
            '/api':{
                target:`http://www.baidu.com`, //请求后台接口
                changeOrigin:true, // 允许跨域
                pathRewrite:{
                    '^/api' : '' // 重写请求
                }
            }
        }
    },
}

创建axios实例时，将baseUrl设置为/api

const http = axios.create({
    timeout: 1000 * 1000000,
    withCredentials: true,
    BASE_URL: '/api'
    headers: {
        'Content-Type': 'application/json; charset=utf-8'
    }
})
